*** Please be noticed that this version of qtumjs is 1.3.2
*** After install all required packages

1> Edit Contract.js
- Open node_modules/qtumjs/lib/Contract.js
- Modify the constructor with the code below:

'''
constructor(rpc, info) {
        // modified for reg-test network
        this.rpc = rpc;
        this.info = info;
        this.callMethodsMap = {};
        this.sendMethodsMap = {};
        for (const methodABI of info.abi) {
            const name = methodABI.name;
            // Allow sendToContract only for non-constant methods
            if (!methodABI.constant) {
                this.sendMethodsMap[name] = methodABI;
            }
            this.callMethodsMap[name] = methodABI;
        }

        // modified for reg test network
        this.address = info.networks["4132"].address;
        this.address = this.address.slice(2, this.address.length)
        
        console.log('contract address: ', this.address)
        console.log('contract loaded');
    }
'''

- Add method updateAddress right below the constructor:

'''
// update contract address in case the json format file is changed
    updateAddress(contractAddress){
        this.address = contractAddress;
        console.log('contract address: ', this.address)
    }
'''
